<script>
function login() {
  $.post("/login", {
name: $("#name").val(),
password: $("#password").val() }, function(res) {
if(res.login){
alert("JWT Token:\n\n" + res.token);   
      } else {
alert("\nAuthentication Failed");        
      }
    });
}
